CREATE TABLE IF NOT EXISTS inmuebles (
    Id INTEGER PRIMARY KEY,
    Barrio VARCHAR ,
    Poblacion TEXT ,
    Municipio TEXT ,
    Provincia TEXT NOT NULL,
    Direccion TEXT ,
    Planta INTEGER ,
    Numero TEXT ,
    Puerta TEXT ,
    Latitud REAL NOT NULL,
    Longitud REAL NOT NULL,
    CheckCercaPlaya BOOLEAN ,
    CheckVistasMar BOOLEAN ,
    CheckVistasCiudad BOOLEAN ,
    CheckVistasMontana BOOLEAN ,
    CheckVistasDestacadas BOOLEAN ,
    CheckOrientacionSur BOOLEAN ,
    CheckObraNueva BOOLEAN ,
    CheckLuz BOOLEAN ,
    Metros_Construidos INTEGER ,
    Metros_Utiles INTEGER ,
    Metros_Parcela INTEGER ,
    Operacion TEXT CHECK (Operacion IN ('Venta', 'Alquiler', 'Traspaso')) ,
    Tipo TEXT CHECK (Tipo IN ('Casas o chalets', 'Locales', 'Fincas y solares', 'Pisos', 'Garajes', 'Oficinas', 'Negocios', 'Naves', 'Trasteros', 'Edificios')) NOT NULL,
    Subtipo TEXT CHECK (Subtipo IN ('Independientes', 'Edificables', 'No edificables', 'Adosadas', 'Pareadas', 'Solares', 'Bar', 'Restaurante', 'Hotel', 'Locales comerciales', 'Suelo industrial')) ,
    CheckAtico BOOLEAN ,
    CheckDuplex BOOLEAN ,
    Antiguedad INTEGER ,
    NumAseos INTEGER ,
    CheckAscensor BOOLEAN ,
    NumDormitorios INTEGER ,
    Cocina TEXT CHECK (Cocina IN ('Independiente', 'Amueblada sin electrodomesticos', 'Amueblada con electrodomesticos', 'Americana', 'No tiene cocina')) ,
    NumTerrazas INTEGER ,
    Balcon INTEGER ,
    Metros_Balcon INTEGER ,
    CheckGaraje BOOLEAN ,
    CheckTrastero BOOLEAN ,
    CheckSotano BOOLEAN ,
    CheckPiscina BOOLEAN ,
    CheckJardin BOOLEAN ,
    Metros_Jardin INTEGER ,
    CheckPatio BOOLEAN ,
    Agua_Caliente ENUM ,
    Calefaccion ENUM ,
    CheckAireAcondicionado BOOLEAN ,
    CheckChimenea BOOLEAN ,
    Gastos_Comunidad INTEGER ,
    Precio INTEGER ,
    PrioridadRK BOOLEAN ,
    CheckAlquilerTemporal BOOLEAN ,
    CheckAlquilerOpcionCompra BOOLEAN ,
    CheckAlquilerHabitacion BOOLEAN ,
    CheckAmueblado BOOLEAN ,
    CheckMascotasSi BOOLEAN ,
    CheckLicenciaTuristica BOOLEAN ,
    Estado_General TEXT CHECK (Estado_General IN ('Sencillo', 'Buen estado', 'Nuevo', 'A estrenar', '1ras Calidades', 'A reformar', 'Reformado')) ,
    Certificado_Energetico TEXT CHECK (Certificado_Energetico IN ('EXENTO', 'D', 'B', 'E', 'C', 'F', 'G', 'A', 'EN TRAMITE')) ,
    TipoVentana TEXT CHECK (TipoVentana IN ('Aluminio', 'PVC', 'Climalit', 'Madera')) ,
    EstacionTrenCerca BOOLEAN ,
    EstacionBusCerca BOOLEAN ,
    ParqueCerca TEXT CHECK (ParqueCerca IN ('Parque San Francisco', 'Parque de Invierno', 'Parque del Oeste', 'Parque de Isabel la Catolica', 'Parque de Moreda', 'Parque de Ferrera')) ,
    EsCentro BOOLEAN ,
    UniversidadCerca TEXT CHECK (UniversidadCerca IN ('Campus del Cristo', 'Campus del Milan', 'La Laboral')) ,
    HospitalCerca TEXT CHECK (HospitalCerca IN ('Hospital Universitario Central de Asturias (HUCA)', 'Hospital Jove', 'Hospital Universitario de Cabue√±es')) 
);
